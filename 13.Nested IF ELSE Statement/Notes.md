this is example of nested if else 

if (condition) {
    console.log('Your Text');
}
else if (condition) {
    console.log('Your Text');
    if (condition) {
        console.log('Your Text');
    }
    else {
        console.log('Your Text');
        if (condition) {
            console.log('Your Text')
        }
        else {
            console.log('Your Text')
        }
    }
}
